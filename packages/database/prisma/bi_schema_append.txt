model BiDashboard {
  id          String    @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  tenantId    String    @map("tenant_id") @db.Uuid
  name        String    @db.VarChar(200)
  slug        String    @db.VarChar(100)
  description String?   @db.Text
  type        String    @default("custom") @db.VarChar(30)
  module      String?   @db.VarChar(50)
  isDefault   Boolean   @default(false) @map("is_default")
  isPublic    Boolean   @default(false) @map("is_public")
  layout      Json?
  filters     Json?
  refreshRate Int?      @map("refresh_rate")
  createdBy   String?   @map("created_by") @db.Uuid
  updatedBy   String?   @map("updated_by") @db.Uuid
  deletedAt   DateTime? @map("deleted_at")
  isActive    Boolean   @default(true) @map("is_active")
  createdAt   DateTime  @default(now()) @map("created_at")
  updatedAt   DateTime  @updatedAt @map("updated_at")

  tenant      CoreTenant              @relation(fields: [tenantId], references: [id])
  widgets     BiWidget[]
  permissions BiDashboardPermission[]

  @@unique([tenantId, slug])
  @@index([tenantId])
  @@map("bi_dashboards")
}

model BiWidget {
  id          String   @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  tenantId    String   @map("tenant_id") @db.Uuid
  dashboardId String   @map("dashboard_id") @db.Uuid
  title       String   @db.VarChar(200)
  type        String   @db.VarChar(30)
  module      String?  @db.VarChar(50)
  dataSource  String?  @map("data_source") @db.VarChar(100)
  config      Json?
  positionX   Int      @default(0) @map("position_x")
  positionY   Int      @default(0) @map("position_y")
  width       Int      @default(4)
  height      Int      @default(3)
  refreshRate Int?     @map("refresh_rate")
  isActive    Boolean  @default(true) @map("is_active")
  createdAt   DateTime @default(now()) @map("created_at")
  updatedAt   DateTime @updatedAt @map("updated_at")

  dashboard BiDashboard @relation(fields: [dashboardId], references: [id], onDelete: Cascade)

  @@index([tenantId, dashboardId])
  @@map("bi_widgets")
}

model BiReport {
  id          String    @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  tenantId    String    @map("tenant_id") @db.Uuid
  name        String    @db.VarChar(200)
  description String?   @db.Text
  module      String    @db.VarChar(50)
  entity      String    @db.VarChar(50)
  columns     Json
  filters     Json?
  groupBy     Json?     @map("group_by")
  sortBy      Json?     @map("sort_by")
  chartType   String?   @map("chart_type") @db.VarChar(30)
  chartConfig Json?     @map("chart_config")
  isPublic    Boolean   @default(false) @map("is_public")
  createdBy   String?   @map("created_by") @db.Uuid
  updatedBy   String?   @map("updated_by") @db.Uuid
  deletedAt   DateTime? @map("deleted_at")
  isActive    Boolean   @default(true) @map("is_active")
  createdAt   DateTime  @default(now()) @map("created_at")
  updatedAt   DateTime  @updatedAt @map("updated_at")

  tenant           CoreTenant          @relation(fields: [tenantId], references: [id])
  scheduledReports BiScheduledReport[]
  exports          BiReportExport[]

  @@index([tenantId])
  @@map("bi_reports")
}

model BiScheduledReport {
  id         String    @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  tenantId   String    @map("tenant_id") @db.Uuid
  reportId   String    @map("report_id") @db.Uuid
  name       String    @db.VarChar(200)
  cronExpr   String    @map("cron_expr") @db.VarChar(100)
  format     String    @default("pdf") @db.VarChar(20)
  recipients Json
  subject    String?   @db.VarChar(300)
  message    String?   @db.Text
  isActive   Boolean   @default(true) @map("is_active")
  lastRunAt  DateTime? @map("last_run_at")
  nextRunAt  DateTime? @map("next_run_at")
  createdBy  String?   @map("created_by") @db.Uuid
  createdAt  DateTime  @default(now()) @map("created_at")
  updatedAt  DateTime  @updatedAt @map("updated_at")

  tenant  CoreTenant       @relation(fields: [tenantId], references: [id])
  report  BiReport         @relation(fields: [reportId], references: [id])
  exports BiReportExport[]

  @@index([tenantId])
  @@map("bi_scheduled_reports")
}

model BiReportExport {
  id            String    @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  tenantId      String    @map("tenant_id") @db.Uuid
  reportId      String    @map("report_id") @db.Uuid
  scheduledId   String?   @map("scheduled_id") @db.Uuid
  format        String    @db.VarChar(20)
  status        String    @default("pending") @db.VarChar(20)
  rowCount      Int?      @map("row_count")
  fileSizeBytes Int?      @map("file_size_bytes")
  fileUrl       String?   @map("file_url") @db.VarChar(500)
  errorMessage  String?   @map("error_message") @db.Text
  requestedBy   String?   @map("requested_by") @db.Uuid
  completedAt   DateTime? @map("completed_at")
  expiresAt     DateTime? @map("expires_at")
  createdAt     DateTime  @default(now()) @map("created_at")
  updatedAt     DateTime  @updatedAt @map("updated_at")

  report    BiReport           @relation(fields: [reportId], references: [id])
  scheduled BiScheduledReport? @relation(fields: [scheduledId], references: [id])

  @@index([tenantId])
  @@map("bi_report_exports")
}

model BiKpiSnapshot {

  kpiKey        String   @map("kpi_key") @db.VarChar(100)

  @@unique([tenantId, module, kpiKey, period, periodDate])

}

model BiDashboardPermission {
  id          String   @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  tenantId    String   @map("tenant_id") @db.Uuid
  dashboardId String   @map("dashboard_id") @db.Uuid
  userId      String   @map("user_id") @db.Uuid
  canView     Boolean  @default(true) @map("can_view")
  canEdit     Boolean  @default(false) @map("can_edit")
  canShare    Boolean  @default(false) @map("can_share")
  createdAt   DateTime @default(now()) @map("created_at")

  dashboard BiDashboard @relation(fields: [dashboardId], references: [id], onDelete: Cascade)

  @@unique([dashboardId, userId])
  @@index([tenantId])
  @@map("bi_dashboard_permissions")
}
